CREATE TABLE `cabecera` (
  `IDOCVTA` int NOT NULL AUTO_INCREMENT,
  `TIPODOC` varchar(3) DEFAULT NULL,
  `NUMERO` int DEFAULT NULL,
  `IDCLIENTE` int DEFAULT NULL,
  `DESCUENTO_TOTAL` decimal(9,4) DEFAULT NULL,
  `SUBTOTAL` decimal(9,4) DEFAULT NULL,
  `MONTOIGV` decimal(9,4) DEFAULT NULL,
  `TOTAL` decimal(9,4) DEFAULT NULL,
  `FECHA_DOC` date DEFAULT NULL,
  `MONEDA` varchar(3) DEFAULT NULL,
  `ESTADO` varchar(3) DEFAULT NULL,
  PRIMARY KEY (`IDOCVTA`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
